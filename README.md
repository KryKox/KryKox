package fr.krykox.lobby;

import java.util.Arrays;

public class Profile {
        
        private String name = "KryKox";
        private int age = 19;
        
        public Profile(String name, int age) {
            this.name = name;
            this.age = age;
        }
    }
    
    public String getName() {
        return name;
    }
    
    public int getAge() {
        return age;
    }
}
